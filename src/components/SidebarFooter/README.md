const SidebarFooterButton = require('./SidebarFooterButton').default;
const renderButtons = () => {
  const buttons = [{
    glyph: 'logo', handler: console.debug
  }, {

  }]

      />
    );
  });
};

  <SidebarFooter
    renderButtons={renderButtons}
